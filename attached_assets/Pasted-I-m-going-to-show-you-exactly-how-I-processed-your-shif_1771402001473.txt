function detectDive(events) {
  const diveKeywords = ["touch", "riser", "dredg", "chipping", "diver"];

  const diveEvents = events.filter(e =>
    diveKeywords.some(k => e.text.toLowerCase().includes(k))
  );

  return diveEvents;
}

4️⃣ Closeout Gate
function evaluateCloseout(day) {
  const gaps = [];

  if (!day.supervisor) gaps.push("Supervisor missing");
  if (!day.standbyDiver) gaps.push("Standby Diver missing");
  if (!day.depthBand) gaps.push("Depth band missing");

  day.dives.forEach(d => {
    if (!d.maxDepth) gaps.push("Dive max depth missing");
    if (!d.gas) gaps.push("Gas not recorded");
    if (!d.commsConfirmed) gaps.push("Comms not confirmed");
  });

  return gaps.length > 0
    ? { status: "NEEDS_INFO", gaps }
    : { status: "PASS" };
}
